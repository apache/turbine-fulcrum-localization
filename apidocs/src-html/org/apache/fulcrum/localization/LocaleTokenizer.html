023import java.util.ArrayList;
024import java.util.Collections;
025import java.util.Iterator;
026import java.util.Locale;
027import java.util.NoSuchElementException;
028import java.util.StringTokenizer;
029
030/**
031 * Parses the HTTP <code>Accept-Language</code> header as per section
032 * 14.4 of RFC 2068 (HTTP 1.1 header field definitions).
033 *
034 * @author <a href="mailto:dlr@collab.net">Daniel Rall</a>
035 * @version $Id$
036 */
037public class LocaleTokenizer
038    implements Iterator<Object>
039{
040    /**
041     * Separates elements of the <code>Accept-Language</code> HTTP
042     * header.
043     */
044    private static final String LOCALE_SEPARATOR = ",";
045
046    /**
047     * Separates locale from quality within elements.
048     */
049    private static final char QUALITY_SEPARATOR = ';';
050
051    /**
052     * The default quality value for an <code>AcceptLanguage</code>
053     * object.
054     */
055    protected static final Float DEFAULT_QUALITY = new Float(1.0f);
056
057    /**
058     * The parsed locales.
059     */
060    private ArrayList<AcceptLanguage> locales = new ArrayList<AcceptLanguage>(3);
061
062    /**
063     * Parses the <code>Accept-Language</code> header.
064     *
065     * @param header The <code>Accept-Language</code> header
066     * (i.e. <code>en, es;q=0.8, zh-TW;q=0.1</code>).
067     */
068    public LocaleTokenizer(String header)
069    {
070        StringTokenizer tok = new StringTokenizer(header, LOCALE_SEPARATOR);
071        while (tok.hasMoreTokens())
072        {
073            AcceptLanguage acceptLang = new AcceptLanguage();
074            String element = tok.nextToken().trim();
075            int index;
076
077            // Record and cut off any quality value that comes after a
078            // semi-colon.
079            if ( (index = element.indexOf(QUALITY_SEPARATOR)) != -1 )
080            {
081                String q = element.substring(index);
082                element = element.substring(0, index);
083                if ( (index = q.indexOf('=')) != -1 )
084                {
085                    try
086                    {
087                        acceptLang.quality =
088                            Float.valueOf(q.substring(index + 1));
089                    }
090                    catch (NumberFormatException useDefault)
091                    {
092                        // ignore
093                    }
094                }
095            }
096
097            element = element.trim();
098
099            // Create a Locale from the language.  A dash may separate the
100            // language from the country.
101            if ( (index = element.indexOf('-')) == -1 )
102            {
103                // No dash means no country.
104                acceptLang.locale = new Locale(element, "");
105            }
106            else
107            {
108                acceptLang.locale = new Locale(element.substring(0, index),
109                                               element.substring(index + 1));
110            }
111
112            locales.add(acceptLang);
113        }
114
115        // Sort by quality in descending order.
116        Collections.sort(locales, Collections.reverseOrder());
117    }
118
119    /**
120     * @return Whether there are more locales.
121     */
122    public boolean hasNext()
123    {
124        return !locales.isEmpty();
125    }
126
127    /**
128     * Creates a <code>Locale</code> from the next element of the
129     * <code>Accept-Language</code> header.
130     *
131     * @return The next highest-rated <code>Locale</code>.
132     * @throws NoSuchElementException No more locales.
133     */
134    public Object next()
135    {
136        if (locales.isEmpty())
137        {
138            throw new NoSuchElementException();
139        }
140        return ((AcceptLanguage) locales.remove(0)).locale;
141    }
142
143    /**
144     * Not implemented.
145     */
146    public final void remove()
147    {
148        throw new UnsupportedOperationException(getClass().getName() +
149                                                " does not support remove()");
150    }
151
152    /**
153     * Struct representing an element of the HTTP
154     * <code>Accept-Language</code> header.
155     */
156    protected static class AcceptLanguage implements Comparable<Object>
157    {
158        /**
159         * The language and country.
160         */
161        Locale locale;
162
163        /**
164         * The quality of our locale (as values approach
165         * <code>1.0</code>, they indicate increased user preference).
166         */
167        Float quality = DEFAULT_QUALITY;
168        
169                @Override
170        public int compareTo(Object acceptLang)
171        {
172                        if ( acceptLang == null )
173                                throw new NullPointerException("AcceptLanguage not found");
174                        
175                        if ( acceptLang instanceof AcceptLanguage )
176                        {
177                                Float q2 = ((AcceptLanguage) acceptLang).quality;
178                                return quality.compareTo( q2 );
179                        } else {
180                                throw new NullPointerException("The object to compare is not the correct type");
181                        }
182                                
183        }
184    }
185}
