023import java.util.Locale;
024import java.util.MissingResourceException;
025import java.util.ResourceBundle;
026
027/**
028 * <p>
029 * Provides localization functionality using the interface provided
030 * by <code>ResourceBundle</code>, plus leverages a "search path"
031 * style traversal of the <code>ResourceBundle</code> objects named by
032 * the <code>locale.default.bundles</code> to discover a value for a
033 * given key.
034 * </p>
035 *
036 * <p>
037 * It is suggested that one handle
038 * <a href="http://www.math.fu-berlin.de/~rene/www/java/tutorial/i18n/message/messageFormat.html">dealing with concatenated messages</a>
039 * using <code>MessageFormat</code> and properties files.
040 * </p>
041 *
042 * @author <a href="mailto:jon@latchkey.com">Jon S. Stevens</a>
043 * @author <a href="mailto:dlr@finemaltcoding.com">Daniel Rall</a>
044 * @author <a href="mailto:leonardr@collab.net">Leonard Richardson</a>
045 * @author <a href="mailto:mcconnell@apache.org">Stephen McConnell</a>
046 * @author <a href="mailto:tv@apache.org">Thomas Vandahl</a>
047 * @version $Id: LocalizationService.java 535465 2007-05-05 06:58:06Z tv $
048 */
049public interface SimpleLocalizationService
050{
051    String ROLE = SimpleLocalizationService.class.getName();
052    String SERVICE_NAME = ROLE;
053
054    /**
055     * Retrieves the default language (as specified in the config
056     * file).
057     * @return the default language
058     */
059    String getDefaultLanguage();
060
061    /**
062     * Retrieves the default country (as specified in the config
063     * file).
064     * 
065     * @return the default country
066     */
067    String getDefaultCountry();
068
069    /**
070     * Retrieves the default Locale (as created from default
071     * language and default country).
072     * 
073     * @return the default locale
074     */
075    Locale getDefaultLocale();
076
077    /**
078     * Retrieves the name of the default bundle (as specified in the
079     * config file), or the first in the list if there are more than
080     * one.
081     * 
082     * @return the default bundle name
083     */
084    String getDefaultBundleName();
085
086    /**
087     * Retrieves the list of names of bundles to search by default for
088     * <code>ResourceBundle</code> keys (as specified in the config
089     * file).
090     *
091     * @return The list of configured bundle names.
092     */
093    String[] getBundleNames();
094
095    /**
096     * Convenience method to get the default <code>ResourceBundle</code>.
097     *
098     * @return A localized <code>ResourceBundle</code>.
099     */
100    ResourceBundle getBundle();
101
102    /**
103     * Returns a ResourceBundle given the bundle name and the default
104     * locale information supplied by the configuration.
105     *
106     * @param bundleName Name of bundle.
107     * @return A localized ResourceBundle.
108     */
109    ResourceBundle getBundle(String bundleName);
110
111    /**
112     * Convenience method to get a ResourceBundle based on name and
113     * Locale.
114     *
115     * @param bundleName Name of bundle.
116     * @param locale A Locale.
117     * @return A localized ResourceBundle.
118     */
119    ResourceBundle getBundle(String bundleName, Locale locale);
120
121    /**
122     * Tries very hard to return a value, looking first in the
123     * specified bundle, then searching list of default bundles
124     * (giving precedence to earlier bundles over later bundles).
125     *
126     * @param bundleName Name of the bundle to look in first.
127     * @param locale Locale to get text for.
128     * @param key Name of the text to retrieve.
129     * @return Localized text.
130     * @throws MissingResourceException if the resource is not found
131     */
132    String getString(String bundleName, Locale locale, String key) throws MissingResourceException;
133    
134    /**
135     * Returns the value for the key in the default bundle and the default locale.
136     * 
137     * @param key The key to retrieve the value for.
138     * @return The value mapped to the key.
139     */
140    String getString(String key);
141
142    /**
143     * This method sets the name of the defaultBundle.
144     *
145     * @param defaultBundle Name of default bundle.
146     */
147    void setBundle(String defaultBundle);
148
149    /**
150     * Formats a localized value using the provided object.
151     *
152     * @param bundleName The bundle in which to look for the localizable text.
153     * @param locale The locale for which to format the text.
154     * @param key The identifier for the localized text to retrieve,
155     * @param arg1 The object to use as {0} when formatting the localized text.
156     * @return Formatted localized text.
157     * @see #format(String, Locale, String, Object[])
158     */
159    String format(String bundleName, Locale locale,
160                         String key, Object arg1);
161
162    /**
163     * Formats a localized value using the provided objects.
164     *
165     * @param bundleName The bundle in which to look for the localizable text.
166     * @param locale The locale for which to format the text.
167     * @param key The identifier for the localized text to retrieve,
168     * @param arg1 The object to use as {0} when formatting the localized text.
169     * @param arg2 The object to use as {1} when formatting the localized text.
170     * 
171     * @return Formatted localized text.
172     * 
173     * @see #format(String, Locale, String, Object[])
174     */
175    String format(String bundleName, Locale locale,
176                         String key, Object arg1, Object arg2);
177
178    /**
179     * Formats a localized value using the provided objects.
180     *
181     * @param bundleName The bundle in which to look for the localizable text.
182     * @param locale The locale for which to format the text.
183     * @param key The identifier for the localized text to retrieve,
184     * @param args The objects to use as {0}, {1}, etc. when
185     *             formatting the localized text.
186     * @return Formatted localized text.
187     */
188    String format(String bundleName, Locale locale,
189                         String key, Object[] args);
190}
