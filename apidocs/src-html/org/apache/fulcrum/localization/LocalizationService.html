024import java.util.Locale;
025import java.util.ResourceBundle;
026
027import javax.servlet.http.HttpServletRequest;
028
029/**
030 * <p>
031 * Provides localization functionality using the interface provided
032 * by <code>ResourceBundle</code>, plus leverages a "search path"
033 * style traversal of the <code>ResourceBundle</code> objects named by
034 * the <code>locale.default.bundles</code> to discover a value for a
035 * given key.
036 * </p>
037 *
038 * <p>
039 * It is suggested that one handle
040 * <a href="http://www.math.fu-berlin.de/~rene/www/java/tutorial/i18n/message/messageFormat.html">dealing with concatenated messages</a>
041 * using <code>MessageFormat</code> and properties files.
042 * </p>
043 *
044 * @author <a href="mailto:jon@latchkey.com">Jon S. Stevens</a>
045 * @author <a href="mailto:dlr@finemaltcoding.com">Daniel Rall</a>
046 * @author <a href="mailto:leonardr@collab.net">Leonard Richardson</a>
047 * @author <a href="mailto:mcconnell@apache.org">Stephen McConnell</a>
048 * @author <a href="mailto:tv@apache.org">Thomas Vandahl</a>
049 * @version $Id$
050 */
051public interface LocalizationService extends SimpleLocalizationService
052{
053    String ROLE = LocalizationService.class.getName();
054    String SERVICE_NAME = ROLE;
055
056    /**
057     * A constant for the HTTP <code>Accept-Language</code> header.
058     */
059    String ACCEPT_LANGUAGE = "Accept-Language";
060
061    /**
062     * Convenience method to get a ResourceBundle based on name and
063     * HTTP <code>Accept-Language</code> header.
064     *
065     * @param bundleName Name of bundle.
066     * @param languageHeader A String with the language header.
067     * @return A localized ResourceBundle.
068     */
069    ResourceBundle getBundle(String bundleName, String languageHeader);
070
071    /**
072     * Convenience method to get a ResourceBundle based on HTTP
073     * Accept-Language header in HttpServletRequest.
074     *
075     * @param req The HTTP request to parse the
076     * <code>Accept-Language</code> of.
077     * @return A localized ResourceBundle.
078     */
079    ResourceBundle getBundle(HttpServletRequest req);
080
081    /**
082     * Convenience method to get a <code>ResourceBundle</code> based
083     * on name and HTTP <code>Accept-Language</code> header from a
084     * <code>HttpServletRequest</code>.
085     *
086     * @param bundleName Name of bundle.
087     * @param req The HTTP request to parse the
088     * <code>Accept-Language</code> of.
089     * @return A localized ResourceBundle.
090     */
091    ResourceBundle getBundle(String bundleName, HttpServletRequest req);
092
093    /**
094     * Attempts to pull the <code>Accept-Language</code> header out of
095     * the <code>HttpServletRequest</code> object and then parse it.
096     * If the header is not present, it will return a
097     * <code>null</code> <code>Locale</code>.
098     *
099     * @param req The HTTP request to parse the
100     * <code>Accept-Language</code> of.
101     * @return The parsed locale.
102     */
103    Locale getLocale(HttpServletRequest req);
104
105    /**
106     * Parses the <code>Accept-Language</code> header and attempts to
107     * create a <code>Locale</code> from it.
108     *
109     * @param languageHeader The language header (i.e. <code>en, es;q=0.8,
110     * zh-TW;q=0.1</code>), or <code>null</code> for the locale
111     * corresponding to the default language and country.
112     * @return The parsed locale, or a locale corresponding to the
113     * language and country defaults.
114     */
115    Locale getLocale(String languageHeader);
116}
