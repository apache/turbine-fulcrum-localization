022import java.util.Locale;
023import java.util.ResourceBundle;
024
025import javax.servlet.http.HttpServletRequest;
026
027import org.apache.commons.lang3.StringUtils;
028
029/**
030 * <p>
031 * This class is the single point of access to all localization
032 * resources.  It caches different ResourceBundles for different
033 * Locales.
034 * </p>
035 *
036 * <p>Usage example:</p>
037 *
038 * <pre>
039 * <code>
040 * LocalizationService ls = (LocalizationService) TurbineServices
041 *     .getInstance().getService(LocalizationService.SERVICE_NAME);
042 * </code>
043 * </pre>
044 *
045 * <p>
046 * Then call {@link #getString(String, Locale, String)}, or one of
047 * four methods to retrieve a ResourceBundle:
048 * </p>
049 *
050 * <ul>
051 * <li>getBundle("MyBundleName")</li>
052 * <li>getBundle("MyBundleName", httpAcceptLanguageHeader)</li>
053 * <li>etBundle("MyBundleName", HttpServletRequest)</li>
054 * <li>getBundle("MyBundleName", Locale)</li>
055 * <li>etc.</li>
056 * </ul>
057 *
058 * @author <a href="mailto:jm@mediaphil.de">Jonas Maurus</a>
059 * @author <a href="mailto:jon@latchkey.com">Jon S. Stevens</a>
060 * @author <a href="mailto:novalidemail@foo.com">Frank Y. Kim</a>
061 * @author <a href="mailto:dlr@finemaltcoding.com">Daniel Rall</a>
062 * @author <a href="mailto:leonardr@collab.net">Leonard Richardson</a>
063 * @author <a href="mailto:mcconnell@apache.org">Stephen McConnell</a>
064 * @author <a href="mailto:tv@apache.org">Thomas Vandahl</a>
065 * @version $Id$
066 * avalon.component name="localization" lifestyle="singleton"
067 * avalon.service type="org.apache.fulcrum.localization.LocalizationService"
068 */
069public class DefaultLocalizationService
070    extends SimpleLocalizationServiceImpl
071    implements LocalizationService
072{
073    /**
074     * Creates a new instance.
075     */
076    public DefaultLocalizationService()
077    {
078        super();
079    }
080
081    /**
082     * This method returns a ResourceBundle given the bundle name and
083     * the Locale information supplied in the HTTP "Accept-Language"
084     * header.
085     * 
086     * @see org.apache.fulcrum.localization.LocalizationService#getBundle(java.lang.String, java.lang.String)
087     *
088     * @param bundleName Name of bundle.
089     * @param languageHeader A String with the language header.
090     * @return A localized ResourceBundle.
091     */
092    public ResourceBundle getBundle(String bundleName, String languageHeader)
093    {
094        return getBundle(bundleName, getLocale(languageHeader));
095    }
096
097    /**
098     * This method returns a ResourceBundle given the Locale
099     * information supplied in the HTTP "Accept-Language" header which
100     * is stored in HttpServletRequest.
101     *
102     * @param req HttpServletRequest.
103     * @return A localized ResourceBundle.
104     */
105    public ResourceBundle getBundle(HttpServletRequest req)
106    {
107        return getBundle(getDefaultBundleName(), getLocale(req));
108    }
109
110    /**
111     * @see org.apache.fulcrum.localization.LocalizationService#getBundle(java.lang.String, javax.servlet.http.HttpServletRequest)
112     * 
113     * This method returns a ResourceBundle given the bundle name and
114     * the Locale information supplied in the HTTP "Accept-Language"
115     * header which is stored in HttpServletRequest.
116     *
117     * @param bundleName Name of the bundle to use if the request's
118     * locale cannot be resolved.
119     * @param req HttpServletRequest.
120     * @return A localized ResourceBundle.
121     */
122    public ResourceBundle getBundle(String bundleName, HttpServletRequest req)
123    {
124        return getBundle(bundleName, getLocale(req));
125    }
126
127    /* (non-Javadoc)
128     * @see org.apache.fulcrum.localization.LocalizationService#getLocale(javax.servlet.http.HttpServletRequest)
129     */
130    public Locale getLocale(HttpServletRequest req)
131    {
132        return getLocale(req.getHeader(ACCEPT_LANGUAGE));
133        // (JSS) Backed out this change because Tomcat seems to be returning
134        //       the wrong result and things just are not working.
135        //        Locale l = req.getLocale();
136        //        return (l != null ? l : getLocale(req.getHeader(ACCEPT_LANGUAGE)));
137    }
138
139    /* (non-Javadoc)
140     * @see org.apache.fulcrum.localization.LocalizationService#getLocale(java.lang.String)
141     */
142    public Locale getLocale(String header)
143    {
144        if (StringUtils.isNotEmpty(header))
145        {
146            LocaleTokenizer tok = new LocaleTokenizer(header);
147            if (tok.hasNext())
148            {
149                return (Locale) tok.next();
150            }
151        }
152        
153        // Couldn't parse locale.
154        return getDefaultLocale();
155    }
156}
