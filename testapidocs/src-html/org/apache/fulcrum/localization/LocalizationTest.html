047    {

049    }

051    
052    /**
053     * Test localization
054     * @throws Exception generic exception
055     */
056    @Test
057    public void testLocalization() throws Exception
058    {
059        // Test retrieval of text using multiple default bundles
060        String s = localizationService.getString(null, null, "key1");
061        assertEquals("value1", s, "Unable to retrieve localized text for locale: default");
062        
063        s = localizationService.getString(null, new Locale("en", "US"), "key2");
064        assertEquals("value2", s, "Unable to retrieve localized text for locale: en-US");
065        
066        s = localizationService.getString("org.apache.fulcrum.localization.BarBundle", new Locale("ko", "KR"), "key3");
067        assertEquals(s, "[ko] value3", "Unable to retrieve localized text for locale: ko-KR");
068        
069        try
070        {
071            localizationService.getString("DoesNotExist", new Locale("ko", ""), "key1");
072            fail();
073        }
074        catch (MissingResourceException expectedFailure)
075        {
076            // Asked for resource bundle which does not exist.
077        }
078        catch( Throwable e )
079        {
080            // should not happen
081            fail();
082        }
083        
084        // When a locale is used which cannot be produced for a given
085        // bundle, fall back to the default locale.
086        s = localizationService.getString(null, new Locale("ko", "KR"), "key4");
087        assertEquals(s, "value4", "Unable to retrieve localized text for locale: default");
088        
089        try
090        {
091            localizationService.getString(null, null, "NoSuchKey");
092            fail();
093        }
094        catch (MissingResourceException expectedFailure)
095        {
096            // Asked for key from default bundle which does not exist,
097        }
098    }
099

107        String key3 = "key3";

109        String key4 = "key4";

113        assertEquals(value2, localizationService.getString(key2));
