24  import java.util.Locale;
25  import java.util.ResourceBundle;
26  
27  import javax.servlet.http.HttpServletRequest;
28  
29  /**
30   * <p>
31   * Provides localization functionality using the interface provided
32   * by <code>ResourceBundle</code>, plus leverages a "search path"
33   * style traversal of the <code>ResourceBundle</code> objects named by
34   * the <code>locale.default.bundles</code> to discover a value for a
35   * given key.
36   * </p>
37   *
38   * <p>
39   * It is suggested that one handle
40   * <a href="http://www.math.fu-berlin.de/~rene/www/java/tutorial/i18n/message/messageFormat.html">dealing with concatenated messages</a>
41   * using <code>MessageFormat</code> and properties files.
42   * </p>
43   *
44   * @author <a href="mailto:jon@latchkey.com">Jon S. Stevens</a>
45   * @author <a href="mailto:dlr@finemaltcoding.com">Daniel Rall</a>
46   * @author <a href="mailto:leonardr@collab.net">Leonard Richardson</a>
47   * @author <a href="mailto:mcconnell@apache.org">Stephen McConnell</a>
48   * @author <a href="mailto:tv@apache.org">Thomas Vandahl</a>
49   * @version $Id$
50   */
51  public interface LocalizationService extends SimpleLocalizationService
52  {
53      String ROLE = LocalizationService.class.getName();
54      String SERVICE_NAME = ROLE;
55  
56      /**
57       * A constant for the HTTP <code>Accept-Language</code> header.
58       */
59      String ACCEPT_LANGUAGE = "Accept-Language";
60  
61      /**
62       * Convenience method to get a ResourceBundle based on name and
63       * HTTP <code>Accept-Language</code> header.
64       *
65       * @param bundleName Name of bundle.
66       * @param languageHeader A String with the language header.
67       * @return A localized ResourceBundle.
68       */
69      ResourceBundle getBundle(String bundleName, String languageHeader);
70  
71      /**
72       * Convenience method to get a ResourceBundle based on HTTP
73       * Accept-Language header in HttpServletRequest.
74       *
75       * @param req The HTTP request to parse the
76       * <code>Accept-Language</code> of.
77       * @return A localized ResourceBundle.
78       */
79      ResourceBundle getBundle(HttpServletRequest req);
80  
81      /**
82       * Convenience method to get a <code>ResourceBundle</code> based
83       * on name and HTTP <code>Accept-Language</code> header from a
84       * <code>HttpServletRequest</code>.
85       *
86       * @param bundleName Name of bundle.
87       * @param req The HTTP request to parse the
88       * <code>Accept-Language</code> of.
89       * @return A localized ResourceBundle.
90       */
91      ResourceBundle getBundle(String bundleName, HttpServletRequest req);
92  
93      /**
94       * Attempts to pull the <code>Accept-Language</code> header out of
95       * the <code>HttpServletRequest</code> object and then parse it.
96       * If the header is not present, it will return a
97       * <code>null</code> <code>Locale</code>.
98       *
99       * @param req The HTTP request to parse the
100      * <code>Accept-Language</code> of.
101      * @return The parsed locale.
102      */
103     Locale getLocale(HttpServletRequest req);
104 
105     /**
106      * Parses the <code>Accept-Language</code> header and attempts to
107      * create a <code>Locale</code> from it.
108      *
109      * @param languageHeader The language header (i.e. <code>en, es;q=0.8,
110      * zh-TW;q=0.1</code>), or <code>null</code> for the locale
111      * corresponding to the default language and country.
112      * @return The parsed locale, or a locale corresponding to the
113      * language and country defaults.
114      */
115     Locale getLocale(String languageHeader);
116 }
